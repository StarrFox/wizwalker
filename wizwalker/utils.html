 24async def async_sorted(iterable, /, *, key=None, reverse=False):
 25    """
 26    sorted but key function is awaited
 27    """
 28    if key is None:
 29        return sorted(iterable, reverse=reverse)
 30
 31    evaluated = {}
 32
 33    for item in iterable:
 34        evaluated[item] = await key(item)
 35
 36    return [
 37        i[0] for i in sorted(evaluated.items(), key=lambda it: it[1], reverse=reverse)
 38    ]

 41class XYZ:
 42    def __init__(self, x: float, y: float, z: float):
 43        self.x = x
 44        self.y = y
 45        self.z = z
 46
 47    def __sub__(self, other):
 48        return self.distance(other)
 49
 50    def __str__(self):
 51        return f"<XYZ ({self.x}, {self.y}, {self.z})>"
 52
 53    def __repr__(self):
 54        return str(self)
 55
 56    def __iter__(self):
 57        return iter((self.x, self.y, self.z))
 58
 59    def distance(self, other):
 60        """
 61        Calculate the distance between two points
 62        this does not account for z axis
 63        """
 64        if not isinstance(other, type(self)):
 65            raise ValueError(
 66                f"Can only calculate distance between instances of {type(self)} not {type(other)}"
 67            )
 68
 69        return math.dist((self.x, self.y), (other.x, other.y))
 70
 71    def yaw(self, other):
 72        """Calculate perfect yaw to reach another xyz"""
 73        if not isinstance(other, type(self)):
 74            raise ValueError(
 75                f"Can only calculate yaw between instances of {type(self)} not {type(other)}"
 76            )
 77
 78        return calculate_perfect_yaw(self, other)
 79
 80    def relative_yaw(self, *, x: float = None, y: float = None):
 81        """Calculate relative yaw to reach another x and/or y relative to current"""
 82        if x is None:
 83            x = self.x
 84        if y is None:
 85            y = self.y
 86
 87        other = type(self)(x, y, self.z)
 88        return self.yaw(other)

 91class Rectangle:
 92    def __init__(self, x1: int, y1: int, x2: int, y2: int):
 93        self.x1 = x1
 94        self.y1 = y1
 95        self.x2 = x2
 96        self.y2 = y2
 97
 98    def __str__(self):
 99        return f"<Rectangle ({self.x1}, {self.y1}, {self.x2}, {self.y2})>"
100
101    def __repr__(self):
102        return str(self)
103
104    def __iter__(self):
105        return iter((self.x1, self.x2, self.y1, self.y2))
106
107    def scale_to_client(self, parents: List["Rectangle"], factor: float) -> "Rectangle":
108        """
109        Scale this rectangle base on parents and a scale factor
110
111        Args:
112            parents: List of other rectangles
113            factor: Factor to scale by
114
115        Returns:
116            The scaled rectangle
117        """
118        x1_sum = self.x1
119        y1_sum = self.y1
120
121        for rect in parents:
122            x1_sum += rect.x1
123            y1_sum += rect.y1
124
125        converted = Rectangle(
126            int(x1_sum * factor),
127            int(y1_sum * factor),
128            int(((self.x2 - self.x1) * factor) + (x1_sum * factor)),
129            int(((self.y2 - self.y1) * factor) + (y1_sum * factor)),
130        )
131
132        return converted
133
134    def center(self):
135        """
136        Get the center point of this rectangle
137
138        Returns:
139            The center point
140        """
141        center_x = ((self.x2 - self.x1) // 2) + self.x1
142        center_y = ((self.y2 - self.y1) // 2) + self.y1
143
144        return center_x, center_y
145
146    def paint_on_screen(self, window_handle: int, *, rgb: tuple = (255, 0, 0)):
147        """
148        Paint this rectangle to the screen for debugging
149
150        Args:
151            rgb: Red, green, blue tuple to define the color of the rectangle
152            window_handle: Handle to the window to paint the rectangle on
153        """
154        paint_struct = PAINTSTRUCT()
155        # https://docs.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-getdc
156        device_context = user32.GetDC(window_handle)
157        brush = gdi32.CreateSolidBrush(ctypes.wintypes.RGB(*rgb))
158
159        user32.BeginPaint(window_handle, ctypes.byref(paint_struct))
160
161        # left, top = top left corner; right, bottom = bottom right corner
162        draw_rect = ctypes.wintypes.RECT()
163        draw_rect.left = self.x1
164        draw_rect.top = self.y1
165        draw_rect.right = self.x2
166        draw_rect.bottom = self.y2
167
168        # https://docs.microsoft.com/en-us/windows/win32/api/wingdi/nf-wingdi-createrectrgnindirect
169        region = gdi32.CreateRectRgnIndirect(ctypes.byref(draw_rect))
170        # https://docs.microsoft.com/en-us/windows/win32/api/wingdi/nf-wingdi-fillrgn
171        gdi32.FillRgn(device_context, region, brush)
172
173        user32.EndPaint(window_handle, ctypes.byref(paint_struct))
174        user32.ReleaseDC(window_handle, device_context)
175        gdi32.DeleteObject(brush)
176        gdi32.DeleteObject(region)

319def calculate_perfect_yaw(current_xyz: XYZ, target_xyz: XYZ) -> float:
320    """
321    Calculates the perfect yaw to reach an xyz in a stright line
322
323    Args:
324        current_xyz: Starting position xyz
325        target_xyz: Ending position xyz
326    """
327    target_line = math.dist(
328        (current_xyz.x, current_xyz.y), (target_xyz.x, target_xyz.y)
329    )
330    origin_line = math.dist(
331        (current_xyz.x, current_xyz.y), (current_xyz.x, current_xyz.y - 1)
332    )
333    target_to_origin_line = math.dist(
334        (target_xyz.x, target_xyz.y), (current_xyz.x, current_xyz.y - 1)
335    )
336    # target_angle = math.cos(origin_line / target_line)
337    target_angle = math.acos(
338        (pow(target_line, 2) + pow(origin_line, 2) - pow(target_to_origin_line, 2))
339        / (2 * origin_line * target_line)
340    )
341
342    if target_xyz.x > current_xyz.x:
343        # outside
344        target_angle_degres = math.degrees(target_angle)
345        perfect_yaw = math.radians(360 - target_angle_degres)
346    else:
347        # inside
348        perfect_yaw = target_angle
349
350    return perfect_yaw

354async def wait_for_value(
355    coro, want, sleep_time: float = 0.5, *, ignore_errors: bool = True
356):
357    """
358    Wait for a awaitable to return a value
359
360    Args:
361        coro: awaitable to wait for
362        want: Value wanted
363        sleep_time: Time between calls
364        ignore_errors: If errors should be ignored
365    """
366    while True:
367        try:
368            now = await coro()
369            if now == want:
370                return now
371
372        except Exception as e:
373            if ignore_errors:
374                await asyncio.sleep(sleep_time)
375
376            else:
377                raise e

397async def maybe_wait_for_value_with_timeout(
398    coro,
399    sleep_time: float = 0.5,
400    *,
401    value: Any = None,
402    timeout: Optional[float] = None,
403    ignore_exceptions: bool = True,
404    inverse_value: bool = False,
405):
406    possible_exception = None
407
408    async def _inner():
409        nonlocal possible_exception
410
411        while True:
412            try:
413                res = await coro()
414                if value is not None and inverse_value and res != value:
415                    return res
416
417                elif value is not None and not inverse_value and res == value:
418                    return res
419
420                elif value is None and inverse_value and res is not None:
421                    return res
422
423                else:
424                    return res
425
426            except Exception as e:
427                if ignore_exceptions:
428                    possible_exception = e
429                    await asyncio.sleep(sleep_time)
430
431                else:
432                    raise e
433
434            await asyncio.sleep(sleep_time)
435
436    try:
437        return await asyncio.wait_for(_inner(), timeout)
438    except asyncio.TimeoutError:
439        raise ExceptionalTimeout(
440            f"Timed out waiting for coro {coro.__name__}", possible_exception
441        )

625def get_windows_from_predicate(predicate: Callable) -> list:
626    """
627    Get all windows that match a predicate
628
629    Args:
630        predicate: the predicate windows should pass
631
632    Examples:
633        .. code-block:: py
634
635            def predicate(window_handle):
636                if window_handle == 0:
637                    # handle will be returned
638                    return True
639                else:
640                    # handle will not be returned
641                    return False
642    """
643    handles = []
644
645    def callback(handle, _):
646        if predicate(handle):
647            handles.append(handle)
648
649        # iterate all windows, (True)
650        return 1
651
652    enumwindows_func_type = ctypes.WINFUNCTYPE(
653        ctypes.c_bool,
654        ctypes.c_int,
655        ctypes.POINTER(ctypes.c_int),
656    )
657
658    callback = enumwindows_func_type(callback)
659    user32.EnumWindows(callback, 0)
660
661    return handles

665def pharse_template_id_file(file_data: bytes) -> dict:
666    """
667    Pharse a template id file's data
668    """
669    if not file_data.startswith(b"BINd"):
670        raise RuntimeError("No BINd id string")
671
672    data = zlib.decompress(file_data[0xD:])
673
674    total_size = len(data)
675    data = io.BytesIO(data)
676
677    data.seek(0x24)
678
679    out = {}
680    while data.tell() < total_size:
681        size = ord(data.read(1)) // 2
682
683        string = data.read(size).decode()
684        data.read(8)  # unknown bytes
685
686        # Little endian int
687        entry_id = struct.unpack("<i", data.read(4))[0]
688
689        data.read(0x10)  # next entry
690
691        out[entry_id] = string
692
693    return out

696def pharse_node_data(file_data: bytes) -> dict:
697    """
698    Converts data into a dict of node nums to points
699    """
700    entry_start = b"\xFE\xDB\xAE\x04"
701
702    node_data = {}
703    # no nodes
704    if len(file_data) == 20:
705        return node_data
706
707    # header
708    file_data = file_data[20:]
709
710    last_start = 0
711    while file_data:
712        start = file_data.find(entry_start, last_start)
713        if start == -1:
714            break
715
716        # fmt: off
717        entry = file_data[start: start + 48 + 2]
718
719        cords_data = entry[16: 16 + (4 * 3)]
720        x = struct.unpack("<f", cords_data[0:4])[0]
721        y = struct.unpack("<f", cords_data[4:8])[0]
722        z = struct.unpack("<f", cords_data[8:12])[0]
723
724        node_num = entry[48: 48 + 2]
725        unpacked_num = struct.unpack("<H", node_num)[0]
726        # fmt: on
727
728        node_data[unpacked_num] = (x, y, z)
729
730    return node_data

320def calculate_perfect_yaw(current_xyz: XYZ, target_xyz: XYZ) -> float:
321    """
322    Calculates the perfect yaw to reach an xyz in a stright line
323
324    Args:
325        current_xyz: Starting position xyz
326        target_xyz: Ending position xyz
327    """
328    target_line = math.dist(
329        (current_xyz.x, current_xyz.y), (target_xyz.x, target_xyz.y)
330    )
331    origin_line = math.dist(
332        (current_xyz.x, current_xyz.y), (current_xyz.x, current_xyz.y - 1)
333    )
334    target_to_origin_line = math.dist(
335        (target_xyz.x, target_xyz.y), (current_xyz.x, current_xyz.y - 1)
336    )
337    # target_angle = math.cos(origin_line / target_line)
338    target_angle = math.acos(
339        (pow(target_line, 2) + pow(origin_line, 2) - pow(target_to_origin_line, 2))
340        / (2 * origin_line * target_line)
341    )
342
343    if target_xyz.x > current_xyz.x:
344        # outside
345        target_angle_degres = math.degrees(target_angle)
346        perfect_yaw = math.radians(360 - target_angle_degres)
347    else:
348        # inside
349        perfect_yaw = target_angle
350
351    return perfect_yaw

666def pharse_template_id_file(file_data: bytes) -> dict:
667    """
668    Pharse a template id file's data
669    """
670    if not file_data.startswith(b"BINd"):
671        raise RuntimeError("No BINd id string")
672
673    data = zlib.decompress(file_data[0xD:])
674
675    total_size = len(data)
676    data = io.BytesIO(data)
677
678    data.seek(0x24)
679
680    out = {}
681    while data.tell() < total_size:
682        size = ord(data.read(1)) // 2
683
684        string = data.read(size).decode()
685        data.read(8)  # unknown bytes
686
687        # Little endian int
688        entry_id = struct.unpack("<i", data.read(4))[0]
689
690        data.read(0x10)  # next entry
691
692        out[entry_id] = string

698    """
699    Converts data into a dict of node nums to points
700    """
701    entry_start = b"\xFE\xDB\xAE\x04"
702
703    node_data = {}
704    # no nodes
705    if len(file_data) == 20:
706        return node_data
707

710

716
717        # fmt: off
718        entry = file_data[start: start + 48 + 2]
719
