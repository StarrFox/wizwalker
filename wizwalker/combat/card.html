 24    async def cast(
 25        self,
 26        target: Union["CombatCard", "wizwalker.combat.CombatMember", None],
 27        *,
 28        sleep_time: Optional[float] = 1.0,
 29        debug_paint: bool = False,
 30    ):
 31        """
 32        Cast this Card on another Card; a Member or with no target
 33
 34        Args:
 35            target: Card, Member, or None if there is no target
 36            sleep_time: How long to sleep after enchants and between multicasts or None for no sleep
 37            debug_paint: If the card should be highlighted before clicking
 38        """
 39        if isinstance(target, CombatCard):
 40            cards_len_before = len(await self.combat_handler.get_cards())
 41
 42            await self.combat_handler.client.mouse_handler.click_window(
 43                self._spell_window
 44            )
 45
 46            await asyncio.sleep(sleep_time)
 47
 48            await self.combat_handler.client.mouse_handler.set_mouse_position_to_window(
 49                target._spell_window
 50            )
 51
 52            await asyncio.sleep(sleep_time)
 53
 54            if debug_paint:
 55                await target._spell_window.debug_paint()
 56
 57            await self.combat_handler.client.mouse_handler.click_window(
 58                target._spell_window
 59            )
 60
 61            # wait until card number goes down
 62            while len(await self.combat_handler.get_cards()) > cards_len_before:
 63                await asyncio.sleep(0.1)
 64
 65            # wiz can't keep up with how fast we can cast
 66            if sleep_time is not None:
 67                await asyncio.sleep(sleep_time)
 68
 69        elif target is None:
 70            await self.combat_handler.client.mouse_handler.click_window(
 71                self._spell_window
 72            )
 73            # we don't need to sleep because nothing will be casted after
 74
 75        else:
 76            await self.combat_handler.client.mouse_handler.click_window(
 77                self._spell_window
 78            )
 79
 80            # see above
 81            if sleep_time is not None:
 82                await asyncio.sleep(sleep_time)
 83
 84            await self.combat_handler.client.mouse_handler.click_window(
 85                await target.get_health_text_window()
 86            )
