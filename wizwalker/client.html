289    async def is_loading(self) -> bool:
290        """
291        If the client is currently in a loading screen
292        (does not apply to character load in)
293        """
294        if not self._is_loading_addr:
295            mov_instruction_addr = await self.hook_handler.pattern_scan(
296                b"\xC6\x05....\x00\xC6\x80.....\x48\x8B",
297                module="WizardGraphicalClient.exe",
298            )
299            # first 2 bytes are the mov instruction and mov type (C6 05)
300            rip_offset = await self.hook_handler.read_typed(
301                mov_instruction_addr + 2, "int"
302            )
303            # 7 is the length of this instruction
304            self._is_loading_addr = mov_instruction_addr + 7 + rip_offset
305
306        # 1 -> can't move (loading) 0 -> can move (not loading)
307        return await self.hook_handler.read_typed(self._is_loading_addr, "bool")

418    async def goto(self, x: float, y: float):
419        """
420        Moves the player to a specific x and y
421
422        Args:
423            x: X to move to
424            y: Y to move to
425        """
426        current_xyz = await self.body.position()
427        # (40 / 100) + 1 = 1.4
428        speed_multiplier = ((await self.client_object.speed_multiplier()) / 100) + 1
429        target_xyz = utils.XYZ(x, y, current_xyz.z)
430        distance = current_xyz - target_xyz
431        move_seconds = distance / (WIZARD_SPEED * speed_multiplier)
432        yaw = utils.calculate_perfect_yaw(current_xyz, target_xyz)
433
434        await self.body.write_yaw(yaw)
435        await utils.timed_send_key(self.window_handle, Keycode.W, move_seconds)

526    async def _teleport_object(
527            self,
528            object_address: int,
529            xyz: XYZ,
530            wait_on_inuse: bool = False,
531            wait_on_inuse_timeout: float = 1.0,
532            purge_on_after_unuser_fixer: bool = False,
533            purge_on_after_unuser_fixer_timeout: float = 0.6,
534    ):
535        if not self.hook_handler._check_if_hook_active(MovementTeleportHook):
536            raise RuntimeError("Movement teleport not active")
537
538        if await self._teleport_helper.should_update():
539            if not wait_on_inuse:
540                raise ValueError("Tried to teleport while should update bool is set")
541
542            await maybe_wait_for_value_with_timeout(
543                self._teleport_helper.should_update,
544                value=False,
545                timeout=wait_on_inuse_timeout,
546            )
547
548        jes = await self._get_je_instruction_forward_backwards()
549
550        await self._teleport_helper.write_target_object_address(object_address)
551        await self._teleport_helper.write_position(xyz)
552        await self._teleport_helper.write_should_update(True)
553
554        for je in jes:
555            await self.hook_handler.write_bytes(je, b"\x90" * 6)
556
557        if purge_on_after_unuser_fixer:
558            try:
559                await maybe_wait_for_value_with_timeout(
560                    self._teleport_helper.should_update,
561                    value=False,
562                    timeout=purge_on_after_unuser_fixer_timeout,
563                )
564            except ExceptionalTimeout:
565                movement_teleport_hook = self.hook_handler._get_hook_by_type(MovementTeleportHook)
566                for je, old_bytes in zip(jes, movement_teleport_hook._old_jes_bytes):
567                    await self.hook_handler.write_bytes(je, old_bytes)
568
569                await self._teleport_helper.write_should_update(False)

668    async def _switch_camera(self, new_camera_address: int, old_camera_address: int):
669        def _pack(address):
670            return struct.pack("<Q", address)
671
672        game_client_address = await self.game_client.read_base_address()
673        packed_game_client_address = _pack(game_client_address)
674
675        packed_new_camera_address = _pack(new_camera_address)
676        packed_old_camera_address = _pack(old_camera_address)
677
678        # fmt: off
679        shellcode = (
680                # setup
681                b"\x50"  # push rax
682                b"\x51"  # push rcx
683                b"\x52"  # push rdx
684                b"\x41\x50"  # push r8
685                b"\x41\x51"  # push r9
686
687                # call set_cam(client, new_cam, ?, cam_swap_fn)
688                b"\x48\xB9" + packed_game_client_address +  # mov rcx, client_addr
689                b"\x48\xBA" + packed_new_camera_address +  # mov rdx, new_cam_addr

528            self,
529            object_address: int,

531            wait_on_inuse: bool = False,
532            wait_on_inuse_timeout: float = 1.0,
