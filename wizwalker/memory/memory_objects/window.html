 63        self, predicate: Callable
 64    ) -> List["DynamicWindow"]:
 65        """
 66        async def my_pred(window) -> bool:

 68              return True

 70            return False
 71
 72        await client.root_window.get_windows_by_predicate(my_pred)

305    async def _read_vector(self, address: int, size: int = 3, data_type: str = "float"):
306        type_str = type_format_dict[data_type].replace("<", "")
307        size_per_type = struct.calcsize(type_str)
308
309        vector_bytes = await self.read_bytes(
310            address, size_per_type * size
311        )
312
313        return struct.unpack("<" + type_str * size, vector_bytes)
